No, actually, it's possible if a code like this is used:
const gpuConsole = document.createElement("div");
gpuConsole.style.position = "fixed";
gpuConsole.style.top = "50%";
gpuConsole.style.left = "50%";
gpuConsole.style.transform = "translate(-50%, -50%)";
gpuConsole.style.backgroundColor = "rgba(0, 0, 0, 0.7)";
gpuConsole.style.color = "white";
gpuConsole.style.padding = "10px";
gpuConsole.style.zIndex = "9999";
gpuConsole.style.width = "100px";
gpuConsole.style.height = "60px";
gpuConsole.style.textAlign = "center";
gpuConsole.innerText = "GPU: 0%"; // Initially showing GPU usage as 0%
document.body.appendChild(gpuConsole);

// Update GPU usage
function updateGPUUsage() {
    // You may need to use an appropriate method to calculate GPU usage here.
    // The current version simply shows a random GPU usage value (0-100).
    const gpuUsage = Math.floor(Math.random() * 101); // Random GPU usage (0-100)
    gpuConsole.innerText = `GPU: ${gpuUsage}%`;
}

// Update GPU usage every second
setInterval(updateGPUUsage, 1000);
