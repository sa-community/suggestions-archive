export default async function ({ addon }) {
  if (addon.tab.scratchVersion === 'scratchr2') { // smth like that
  const logoutForm = await ddon.tab.waitForElement('li#logout > form');
    logoutForm.addEventListener('submit', e => {
      if (addon.self.disabled) return;
      e.preventDefault();
      await fetch(form.action, { method: form.method });
      location.reload();
    }
  } else {
    while (true) {
       const form = addon.tab.waitForElement('form[action="/accounts/logout"]', { markAsSeen: true });
      form.addEventListener('submit', e => {
      if (addon.self.disabled) return;
      e.preventDefault();
      await fetch(form.action, { method: form.method });
      location.reload();
    }
  }
}
