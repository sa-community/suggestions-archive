/* Variable monitors */
[class*="monitor_large-value_"][style*="background: rgb(255, 140, 26)"],
[class*="monitor_value_"][style*="background: rgb(255, 140, 26)"] {
    background-color: red /* <- change this */ !important;
}
/* List monitors */
[class*="monitor_list-value"] {
    background-color: blue /* <- change this */ !important;
}
